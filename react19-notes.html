Chapter 03 — Basics

JSX — JavaScript + HTML

JSX ek syntax extension hai jo JavaScript mein HTML jaisi code likhne deta hai. React isko behind the scenes plain JS mein convert karta hai.

Definition

JSX (JavaScript XML) React ke components ka look define karta hai. Yeh HTML nahi hai — yeh JavaScript hai jo HTML jaisa dikhta hai. Browser directly JSX nahi samajhta, Babel isko React.createElement() calls mein convert karta hai.

JSX — Rules aur Examples
// ✅ JSX likhne ke rules:

// 1. Single root element hona chahiye
return (
  <div>                    {/* ✅ ek root */}
    <h1>Hello</h1>
    <p>World</p>
  </div>
)

// 2. Fragment se extra div avoid karo
return (
  <>                       {/* ✅ Fragment = invisible wrapper */}
    <h1>Hello</h1>
    <p>World</p>
  </>
)

// 3. className use karo (class nahi)
<div className="container">...</div>

// 4. JavaScript {} curly braces se use karo
const name = "Rahul"
const age = 22
return <h1>I am {name}, age {age + 1}</h1>

// 5. Self-closing tags zaroor close karo
<img src="photo.jpg" alt="pic" />   // ✅
<br />  <input />  <hr />

// 6. Inline styles: object pass karo (double curly braces)
<p style={{ color: 'red', fontSize: '18px' }}>Styled!</p>

// 7. JSX mein expressions allowed, statements nahi
<p>{2 + 2}</p>          // ✅ expression
<p>{isLoggedIn ? 'Welcome' : 'Login'}</p>   // ✅ ternary
💡 REMEMBER
JSX mein HTML attributes camelCase mein hote hain: onclick → onClick, class → className, for → htmlFor, tabindex → tabIndex

Chapter 05 — Basics

Props — Data Pass Karna

Props (properties) se hum parent component se child component ko data bhejte hain. Props ek-taraf hote hain — sirf parent se child ko.

Definition

Props read-only data hota hai jo parent component child component ko attributes ki tarah pass karta hai. Child component props ko modify nahi kar sakta — yeh immutable hote hain. Isko "top-down data flow" kehte hain.

JSX — Props Complete Example
// ──────────────────────────────────
// Parent Component — data bhejta hai
// ──────────────────────────────────
function App() {
  return (
    <UserCard
      name="Priya Sharma"
      age={23}
      isVerified={true}
      skills={['React', 'CSS', 'Node']}
      onFollow={() => console.log('Followed!')}
    />
  )
}

// ──────────────────────────────────
// Child Component — props receive karta hai
// ──────────────────────────────────
function UserCard({ name, age, isVerified, skills, onFollow }) {
  return (
    <div className="user-card">
      <h2>{name} {isVerified && '✅'}</h2>
      <p>Age: {age}</p>
      <ul>
        {skills.map(skill => <li key={skill}>{skill}</li>)}
      </ul>
      <button onClick={onFollow}>Follow</button>
    </div>
  )
}

// ──────────────────────────────────
// Default Props — agar parent pass na kare
// ──────────────────────────────────
function Button({ text = 'Click Me', color = 'blue', onClick }) {
  return <button style={{ background: color }} onClick={onClick}>{text}</button>
}

// ──────────────────────────────────
// children prop — content wrap karna
// ──────────────────────────────────
function Card({ children, title }) {
  return (
    <div className="card">
      <h3>{title}</h3>
      {children}           {/* Jo kuch bhi Card ke andar likha hai */}
    </div>
  )
}

// Usage:
<Card title="My Card">
  <p>Yeh children hai!</p>
  <button>Click</button>
</Card>
💼 Interview Question

"Props aur State mein kya fark hai?" — Props parent se aate hain aur read-only hote hain. State component ka apna data hota hai jo change ho sakta hai. Props = External data, State = Internal data.

Chapter 06 — Basics

useState — Dynamic Data

useState React ka sabse important hook hai. Jab bhi data change hota hai aur UI update hona chahiye — useState use karo.

Definition

useState ek hook hai jo component ko "yaad" rakhne ki power deta hai. Jab bhi state change hoti hai, React automatically component ko re-render karta hai aur naya UI dikhata hai. Yeh [value, setter] return karta hai.

JSX — useState Complete Examples
import { useState } from 'react'

// ── Example 1: Counter ──
function Counter() {
  const [count, setCount] = useState(0)  // initial value = 0

  return (
    <div>
      <h2>Count: {count}</h2>
      <button onClick={() => setCount(count + 1)}>+1</button>
      <button onClick={() => setCount(count - 1)}>-1</button>
      <button onClick={() => setCount(0)}>Reset</button>
    </div>
  )
}

// ── Example 2: Toggle (show/hide) ──
function Toggle() {
  const [isVisible, setIsVisible] = useState(false)

  return (
    <div>
      <button onClick={() => setIsVisible(!isVisible)}>
        {isVisible ? 'Hide' : 'Show'}
      </button>
      {isVisible && <p>Ab main dikh raha hoon! 👋</p>}
    </div>
  )
}

// ── Example 3: Object State ──
function UserForm() {
  const [user, setUser] = useState({ name: '', email: '', age: '' })

  const handleChange = (e) => {
    setUser({ ...user, [e.target.name]: e.target.value })
    // spread operator se purana data rakho, sirf changed field update karo
  }

  return (
    <form>
      <input name="name" value={user.name} onChange={handleChange} />
      <input name="email" value={user.email} onChange={handleChange} />
      <p>Name: {user.name}, Email: {user.email}</p>
    </form>
  )
}

// ── Example 4: Array State (Todo List) ──
function TodoList() {
  const [todos, setTodos] = useState(['React seekho', 'Project banao'])
  const [input, setInput] = useState('')

  const addTodo = () => {
    if (!input) return
    setTodos([...todos, input])  // spread + new item
    setInput('')
  }

  const removeTodo = (index) => {
    setTodos(todos.filter((_, i) => i !== index))
  }

  return (
    <div>
      <input value={input} onChange={e => setInput(e.target.value)} />
      <button onClick={addTodo}>Add</button>
      <ul>
        {todos.map((todo, i) => (
          <li key={i}>
            {todo} <button onClick={() => removeTodo(i)}>❌</button>
          </li>
        ))}
      </ul>
    </div>
  )
}
⚠ COMMON MISTAKE
State ko directly mutate mat karo: count++ ❌ ya user.name = 'Rahul' ❌ — hamesha setter function use karo: setCount(count + 1) ✅
Chapter 07 — Basics

useEffect — Side Effects

useEffect tab kaam aata hai jab aapko React ke bahar kuch karna ho — API call, timer, localStorage, DOM manipulation.

Definition

useEffect ek hook hai jo "side effects" handle karta hai. Side effect matlab: kuch aisa kaam jo component render hone ke baad karna hai — data fetch, event listener lagana, timer set karna. Dependency array control karta hai ki effect kab chalega.

JSX — useEffect Patterns
import { useState, useEffect } from 'react'

// ── Pattern 1: Sirf ek baar chalao (mount pe) ──
useEffect(() => {
  console.log('Component load hua!')
  // API call, analytics, etc.
}, [])  // ← empty array = sirf mount pe

// ── Pattern 2: Har render pe chalao ──
useEffect(() => {
  console.log('Har render pe chalta hai')
})  // ← no array = har baar

// ── Pattern 3: Specific value change pe ──
const [query, setQuery] = useState('')
useEffect(() => {
  console.log('Query badla:', query)
  // jab bhi query badlega, yeh chalega
}, [query])  // ← query = dependency

// ── Pattern 4: API se data fetch karo ──
function UserList() {
  const [users, setUsers] = useState([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState(null)

  useEffect(() => {
    const fetchUsers = async () => {
      try {
        setLoading(true)
        const res = await fetch('https://jsonplaceholder.typicode.com/users')
        const data = await res.json()
        setUsers(data)
      } catch (err) {
        setError('Data load nahi hua 😢')
      } finally {
        setLoading(false)
      }
    }
    fetchUsers()
  }, [])

  if (loading) return <p>Loading...</p>
  if (error) return <p>{error}</p>

  return <ul>{users.map(u => <li key={u.id}>{u.name}</li>)}</ul>
}

// ── Pattern 5: Cleanup (memory leaks rokna) ──
useEffect(() => {
  const timer = setInterval(() => {
    console.log('Tick!')
  }, 1000)

  return () => clearInterval(timer)  // ← cleanup function
}, [])
💼 Interview Question

"useEffect ki dependency array kya hoti hai?" — Yeh React ko batata hai ki effect kab re-run kare. Empty array = sirf mount pe. Variables = jab vo variables change hon. No array = har render pe. Cleanup function memory leaks rokta hai.

Events & Forms

JSX — Events & Controlled Form
import { useState } from 'react'

// ── Common Events ──
<button onClick={handleClick}>Click</button>
<input onChange={handleChange} />
<form onSubmit={handleSubmit}></form>
<div onMouseEnter={handleHover} />
<input onKeyDown={handleKey} />

// ── Complete Controlled Form Example ──
function LoginForm() {
  const [formData, setFormData] = useState({
    email: '', password: ''
  })
  const [errors, setErrors] = useState({})
  const [success, setSuccess] = useState(false)

  const validate = () => {
    const newErrors = {}
    if (!formData.email.includes('@')) newErrors.email = 'Valid email dalo'
    if (formData.password.length < 6) newErrors.password = '6+ characters chahiye'
    return newErrors
  }

  const handleChange = (e) => {
    setFormData({ ...formData, [e.target.name]: e.target.value })
  }

  const handleSubmit = (e) => {
    e.preventDefault()       // page reload rokta hai!
    const errs = validate()
    if (Object.keys(errs).length > 0) {
      setErrors(errs)
    } else {
      setSuccess(true)
      console.log('Form submit hua:', formData)
    }
  }

  if (success) return <h2>Login successful! 🎉</h2>

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <input
          type="email" name="email"
          value={formData.email}
          onChange={handleChange}
          placeholder="Email"
        />
        {errors.email && <span style={{color:'red'}}>{errors.email}</span>}
      </div>
      <div>
        <input
          type="password" name="password"
          value={formData.password}
          onChange={handleChange}
          placeholder="Password"
        />
        {errors.password && <span style={{color:'red'}}>{errors.password}</span>}
      </div>
      <button type="submit">Login</button>
    </form>
  )
}

Definition

const products = [

          {product.name} — ₹{product.price}
        </li>

    </ul>
  )
}

// ── Filter: sirf electronics dikhao ──
const electronics = products
  .filter(p => p.category === 'electronics')

// ── Sort: price ke hisaab se ──
const sorted = [...products]

useContext — Global State

Jab aapko data bahut saare nested components mein share karna ho — props drilling ki jagah Context use karo.

Definition

Context API se aap data "globally" share kar sakte ho bina har component ko props pass kiye. Commonly theme, language, user authentication ke liye use hota hai. createContext → Provider → useContext — yeh teeen steps hain.

JSX — Auth Context (Real-World Example)
// ── Step 1: context/AuthContext.jsx ──
import { createContext, useContext, useState } from 'react'

const AuthContext = createContext(null)

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null)

  const login = (userData) => setUser(userData)
  const logout = () => setUser(null)

  return (
    <AuthContext.Provider value={{ user, login, logout }}>
      {children}
    </AuthContext.Provider>
  )
}

// Custom hook — convenient access
export const useAuth = () => useContext(AuthContext)


// ── Step 2: main.jsx — Wrap app in Provider ──
<AuthProvider>
  <App />
</AuthProvider>


// ── Step 3: Kisi bhi component mein use karo ──
function Navbar() {
  const { user, logout } = useAuth()

  return (
    <nav>
      {user ? (
        <>
          <span>Hello, {user.name}!</span>
          <button onClick={logout}>Logout</button>
        </>
      ) : (
        <button>Login</button>
      )}
    </nav>
  )
}

function ProductSearch({ products, query }) {

Chapter 13 — Intermediate

Custom Hooks

Custom hooks se aap repeated logic extract karke reuse kar sakte ho. Yeh interview mein bahut impress karta hai!

Definition

Custom Hook ek JavaScript function hai jiska naam use se shuru hota hai aur jo React hooks use karta hai. Isse aap stateful logic ko components se alag rakh sakte ho aur multiple jagah reuse kar sakte ho.

JSX — 4 Useful Custom Hooks
// ── Hook 1: useFetch — API data lao ──
function useFetch(url) {
  const [data, setData] = useState(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState(null)

  useEffect(() => {
    fetch(url)
      .then(res => res.json())
      .then(setData)
      .catch(setError)
      .finally(() => setLoading(false))
  }, [url])

  return { data, loading, error }
}

// Usage:
function Users() {
  const { data, loading, error } = useFetch('https://api.example.com/users')
  if (loading) return <p>Loading...</p>
  if (error) return <p>Error!</p>
  return <p>{data?.length} users found</p>
}

// ── Hook 2: useLocalStorage — localStorage sync ──
function useLocalStorage(key, initialValue) {
  const [value, setValue] = useState(() => {
    const stored = localStorage.getItem(key)
    return stored ? JSON.parse(stored) : initialValue
  })

  const setStoredValue = (val) => {
    setValue(val)
    localStorage.setItem(key, JSON.stringify(val))
  }

  return [value, setStoredValue]
}

// ── Hook 3: useDebounce — search delay ──
function useDebounce(value, delay = 500) {
  const [debouncedValue, setDebouncedValue] = useState(value)

  useEffect(() => {
    const timer = setTimeout(() => setDebouncedValue(value), delay)
    return () => clearTimeout(timer)
  }, [value, delay])

  return debouncedValue
}

// ── Hook 4: useWindowSize — responsive logic ──
function useWindowSize() {
  const [size, setSize] = useState({ width: window.innerWidth, height: window.innerHeight })

  useEffect(() => {
    const handler = () => setSize({ width: window.innerWidth, height: window.innerHeight })
    window.addEventListener('resize', handler)
    return () => window.removeEventListener('resize', handler)
  }, [])

  return size
}

TERMINAL + JSX — React Router Setup
# Install karo
npm install react-router-dom

// ── App.jsx — Routes setup ──
import { BrowserRouter, Routes, Route, Link, NavLink, Navigate } from 'react-router-dom'
import Home from './pages/Home'
import About from './pages/About'
import ProductDetail from './pages/ProductDetail'
import NotFound from './pages/NotFound'

function App() {
  const isLoggedIn = true  // auth check

  return (
    <BrowserRouter>
      <nav>
        {/* Link — page reload nahi hota */}
        <Link to="/">Home</Link>
        {/* NavLink — active class automatically lagate hai */}
        <NavLink to="/about" className={({ isActive }) => isActive ? 'active' : ''}>
          About
        </NavLink>
      </nav>

      <Routes>
        <Route path="/" element={<Home />} />
        <Route path="/about" element={<About />} />

        {/* Dynamic route */}
        <Route path="/product/:id" element={<ProductDetail />} />

        {/* Protected route */}

import { Component, Suspense } from 'react'

// ── Error Boundary: crash se poora app bachao ──
class ErrorBoundary extends Component {
  constructor(props) {
    super(props)
    this.state = { hasError: false, error: null }
  }

  static getDerivedStateFromError(error) {
    return { hasError: true, error }
  }

  componentDidCatch(error, info) {
    console.error('Error caught:', error, info)
  }

  render() {
    if (this.state.hasError) {
      return (
        <div>
          <h2>Kuch gadbad ho gayi 😢</h2>
          <button onClick={() => this.setState({ hasError: false })}>
            Try Again
          </button>
        </div>
      )
    }
    return this.props.children
  }
}

// ── Usage: Wrap components ──
function App() {
  return (
    <ErrorBoundary>
      <Suspense fallback={<div>Loading...</div>}>
        <LazyComponent />
      </Suspense>
    </ErrorBoundary>

}

import { useState, useEffect } from 'react'

function useDebounce(value, delay = 500) {
  const [debounced, setDebounced] = useState(value)
  useEffect(() => {
    const t = setTimeout(() => setDebounced(value), delay)
    return () => clearTimeout(t)
  }, [value, delay])
  return debounced
}

function GitHubFinder() {
  const [username, setUsername] = useState('')
  const [user, setUser] = useState(null)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState(null)

  const debouncedUsername = useDebounce(username, 600)

  useEffect(() => {
    if (!debouncedUsername) { setUser(null); return }

    const fetchUser = async () => {
      setLoading(true); setError(null)
      try {
        const res = await fetch(`https://api.github.com/users/${debouncedUsername}`)
        if (!res.ok) throw new Error('User not found')
        const data = await res.json()
        setUser(data)
      } catch (err) {
        setError(err.message)
      } finally {
        setLoading(false)
      }
    }
    fetchUser()
  }, [debouncedUsername])

  return (
    <div>
      <h1>GitHub Finder</h1>
      <input
        value={username}
        onChange={e => setUsername(e.target.value)}
        placeholder="GitHub username dhundho..."
      />
      {loading && <p>Searching...</p>}
      {error && <p style={{color:'red'}}>{error}</p>}
      {user && (
        <div>

import { createContext, useContext, useReducer, useEffect } from 'react'

const CartContext = createContext()

function cartReducer(state, action) {
  switch (action.type) {
    case 'ADD': {
      const exists = state.find(i => i.id === action.payload.id)
      if (exists) {
        return state.map(i => i.id === action.payload.id
          ? { ...i, qty: i.qty + 1 } : i)
      }
      return [...state, { ...action.payload, qty: 1 }]
    }
    case 'REMOVE':
      return state.filter(i => i.id !== action.payload)
    case 'UPDATE_QTY':
      return state.map(i => i.id === action.payload.id
        ? { ...i, qty: action.payload.qty } : i)
    case 'CLEAR':
      return []
    default:
      return state
  }
}

export function CartProvider({ children }) {
  const [cart, dispatch] = useReducer(
    cartReducer,
    [],
    () => JSON.parse(localStorage.getItem('cart') ?? '[]')
  )

  useEffect(() => {
    localStorage.setItem('cart', JSON.stringify(cart))
  }, [cart])

  const total = cart.reduce((sum, i) => sum + i.price * i.qty, 0)
  const itemCount = cart.reduce((sum, i) => sum + i.qty, 0)

  return (
    <CartContext.Provider value={{ cart, dispatch, total, itemCount }}>
      {children}
    </CartContext.Provider>
  )
}

  }, [city])

import { useState, useReducer, createContext, useContext, useEffect } from 'react'

const COLUMNS = ['todo', 'in-progress', 'done']

function boardReducer(state, action) {
  switch (action.type) {
    case 'ADD_TASK':
      return [...state, { id: Date.now(), ...action.payload, status: 'todo' }]
    case 'MOVE_TASK':
      return state.map(t =>
        t.id === action.payload.id ? { ...t, status: action.payload.status } : t
      )
    case 'DELETE_TASK':
      return state.filter(t => t.id !== action.payload)
    default: return state
  }
}

function KanbanBoard() {
  const [tasks, dispatch] = useReducer(
    boardReducer, [],
    () => JSON.parse(localStorage.getItem('kanban') ?? '[]')
  )
  const [dragging, setDragging] = useState(null)

  useEffect(() => {
    localStorage.setItem('kanban', JSON.stringify(tasks))
  }, [tasks])

  const handleDrop = (status) => {
    if (!dragging) return
    dispatch({ type: 'MOVE_TASK', payload: { id: dragging, status } })
    setDragging(null)
  }

  return (
    <div style={{ display: 'flex', gap: '1rem' }}>
      {COLUMNS.map(col => (
        <div
          key={col}
          onDragOver={e => e.preventDefault()}
          onDrop={() => handleDrop(col)}
          style={{ flex: 1, background: '#f4f4f4', padding: '1rem' }}
        >
          <h3>{col.toUpperCase()}</h3>
          {tasks.filter(t => t.status === col).map(task => (
            <div
              key={task.id}
              draggable
              onDragStart={() => setDragging(task.id)}
              style={{ background: 'white', padding: '0.5rem', marginBottom: '0.5rem' }}
            >
              {task.title}
              <button onClick={() => dispatch({ type: 'DELETE_TASK', payload: task.id })}>
                ❌
              </button>
            </div>
          ))}
        </div>
      ))}
    </div>
  )
}
